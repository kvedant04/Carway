        // Show notification with specific login method
        function showLoginNotification(method, customMessage = null) {
            if (notificationTimeout) {
                clearTimeout(notificationTimeout);
                notification.classList.remove('show');
                notification.style.display = 'none';
            }

            const notificationIcon = notification.querySelector('.notification-icon');
            const notificationTitle = notification.querySelector('.notification-content h4');

            notification.className = 'notification';
            notification.style.display = 'none';

            if (customMessage) {
                notificationIcon.textContent = customMessage.icon || '👋🏻';
                notificationTitle.textContent = customMessage.title;
                notificationMessage.textContent = customMessage.message;

                if (customMessage.type === 'success') {
                    notification.classList.add('success');
                } else if (customMessage.type === 'error') {
                    notification.classList.add('error');
                } else if (customMessage.type === 'info') {
                    notification.classList.add('info');
                }
            } else {
                switch(method) {
                    case 'google':
                        notification.classList.add('google');
                        notificationIcon.innerHTML = '<img width="24" height="24" src="https://img.icons8.com/color/48/google-logo.png" alt="google-logo"/>';
                        notificationTitle.textContent = 'Google Login Successful';
                        notificationMessage.textContent = 'You\'ve successfully logged in with Google.';
                        break;
                    case 'facebook':
                        notification.classList.add('facebook');
                        notificationIcon.innerHTML = '<img width="24" height="24" src="https://img.icons8.com/fluency-systems-filled/48/facebook-new.png" alt="facebook-new"/>';
                        notificationTitle.textContent = 'Facebook Login Successful';
                        notificationMessage.textContent = 'You\'ve successfully logged in with Facebook.';
                        break;
                    case 'email':
                    default:
                        notification.classList.add('email');
                        notificationIcon.textContent = '👋🏻';
                        notificationTitle.textContent = 'Welcome Back!';

        // Validate email format
        function validateEmail(email) {
            const re = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
            return re.test(String(email).toLowerCase());
        }

        // Validate password strength
        function checkPasswordStrength(password) {
            let strength = 0;

            // Length check
            if (password.length >= 8) strength++;

            // Uppercase check
            if (/[A-Z]/.test(password)) strength++;

            // Number check
            if (/\d/.test(password)) strength++;

            // Special char check
            if (/[^A-Za-z0-9]/.test(password)) strength++;

            return strength;
        }

        // Update password strength indicator
        function updatePasswordStrength(password) {
            const strength = checkPasswordStrength(password);
            passwordStrength.setAttribute('data-strength', strength);

            // Update hint list
            lengthHint.classList.toggle('valid', password.length >= 8);
            uppercaseHint.classList.toggle('valid', /[A-Z]/.test(password));
            numberHint.classList.toggle('valid', /\d/.test(password));
            specialHint.classList.toggle('valid', /[^A-Za-z0-9]/.test(password));
        }

            loginSubmitBtn.disabled = true;

        // Handle car comparison
        function handleCarComparison() {
            const car1 = compareCar1.value;
            const car2 = compareCar2.value;

            if (!car1 || !car2) {
                showNotification('Please select two cars to compare', 'error');
                return;
            }

            if (car1 === car2) {
                showNotification('Please select two different cars to compare', 'error');

        // Function to compare two cars
        function compareCars(car1Id, car2Id) {
            const car1 = carData[car1Id];
            const car2 = carData[car2Id];

            if (!car1 || !car2) {
                showNotification('Invalid car selection', 'error');
                return;
            }

            // Update car info
            document.getElementById('compareCar1Name').textContent = car1.name;
            document.getElementById('compareCar2Name').textContent = car2.name;
            document.getElementById('compareCar1Price').textContent = car1.price;
            document.getElementById('compareCar2Price').textContent = car2.price;
            document.getElementById('compareCar1Image').src = car1.image;
            document.getElementById('compareCar1Image').alt = car1.name;
            document.getElementById('compareCar2Image').src = car2.image;
            document.getElementById('compareCar2Image').alt = car2.name;

            // Generate comparison features
            const comparisonFeatures = document.getElementById('comparisonFeatures');
            comparisonFeatures.innerHTML = '';

            const features = [
                { name: 'Price', key: 'price' },
                { name: 'Engine', key: 'engine' },
                { name: 'Power', key: 'power' },
                { name: 'Torque', key: 'torque' },
                { name: 'Mileage/Range', key: 'mileage' },
                { name: 'Transmission', key: 'transmission' },
                { name: 'Fuel Type', key: 'fuelType' },
                { name: 'Seating Capacity', key: 'seating' },
                { name: 'Key Features', key: 'features' }
            ];

            features.forEach(feature => {
                const row = document.createElement('div');
                row.className = 'comparison-row';

                const featureName = document.createElement('div');
                featureName.className = 'comparison-feature-name';
                featureName.textContent = feature.name;

                const value1 = document.createElement('div');
                value1.className = 'comparison-feature-value';
                value1.textContent = car1[feature.key];

                const value2 = document.createElement('div');
                value2.className = 'comparison-feature-value';
                value2.textContent = car2[feature.key];

                // Highlight better value (simple comparison for demo)
                if (feature.key === 'mileage') {
                    const mileage1 = parseFloat(car1[feature.key]);
                    const mileage2 = parseFloat(car2[feature.key]);
                    if (mileage1 > mileage2) {
                        value1.classList.add('comparison-highlight');
                    } else if (mileage2 > mileage1) {
                        value2.classList.add('comparison-highlight');
                    }
                } else if (feature.key === 'power') {
                    const power1 = parseInt(car1[feature.key]);
                    const power2 = parseInt(car2[feature.key]);
                    if (power1 > power2) {
                        value1.classList.add('comparison-highlight');
                    } else if (power2 > power1) {
                        value2.classList.add('comparison-highlight');
                    }
                }

                row.appendChild(featureName);
                row.appendChild(value1);
                row.appendChild(value2);

                comparisonFeatures.appendChild(row);
            });

            // Update test drive button
            testDriveBtn.onclick = function() {
                showNotification(`Test drive requested for ${car1.name} and ${car2.name}`, 'success');
            };

            // Show comparison results
            document.getElementById('noComparison').style.display = 'none';
            document.getElementById('comparisonData').style.display = 'block';
            comparisonResults.style.display = 'block';

            // Scroll to results
            comparisonResults.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
        }

            // Send message function
            async function sendMessage() {
                const message = chatbotInput.value.trim();
                if (!message) return;

                // Add user message to chat
                addMessage(message, 'user');
                chatbotInput.value = '';

                // Show typing indicator
                chatbotTypingIndicator.style.display = 'block';
                chatbotMessages.scrollTop = chatbotMessages.scrollHeight;

                try {
                    // Get AI response from backend
                    const response = await getAIResponse(message);

                    // Add AI response to chat
                    addMessage(response, 'bot');
                } catch (error) {
                    console.error('Chatbot error:', error);
                    addMessage("I'm having trouble connecting to the AI service. Please try again later.", 'bot');
                } finally {
                    // Hide typing indicator
                    chatbotTypingIndicator.style.display = 'none';
                    chatbotMessages.scrollTop = chatbotMessages.scrollHeight;
                }
            }

            // Get AI response (simulated or real API call)
           async function getAIResponse(message) {
                try {
                    let backendURL;

                    // Detect file:// or localhost
                    if (window.location.hostname === "localhost" || window.location.hostname === "") {
                        backendURL = "http://localhost:3000/chat";
                    } else {
                        backendURL = "https://your-backend-domain/chat"; // replace with your real domain when deployed
                    }

                    const res = await fetch(backendURL, {
                        method: "POST",
                        headers: { "Content-Type": "application/json" },
                        body: JSON.stringify({ message })
                    });

                    if (!res.ok) {
                        throw new Error(`Server error: ${res.status}`);
                    }

                    const data = await res.json();
                    return data.reply || "Sorry, I couldn't get a response right now.";
                } catch (error) {
                    console.error("Gemini API error:", error);
                    return "I'm having trouble connecting to the AI service. Please try again later.";
                }
            }

        // Car Filtering Functionality
        function initializeCarFiltering() {
            if (!searchButton || !filteredCarsContainer) return;

            let currentType = 'new';

            // Set up event listeners for car filtering
            searchButton.addEventListener('click', filterCars);

            if (typeBtns.length > 0) {
                typeBtns.forEach(btn => {
                    btn.addEventListener('click', function() {
                        typeBtns.forEach(b => b.classList.remove('active'));
                        this.classList.add('active');
                        currentType = this.dataset.type;
                        filterCars();
                    });
                });
            }

            // Initial filter
            filterCars();

            function filterCars() {
                const selectedBudget = budgetFilter ? budgetFilter.value : 'all';
                const selectedBodyType = bodyTypeFilter ? bodyTypeFilter.value : 'all';
                const selectedFuelType = fuelTypeFilter ? fuelTypeFilter.value : 'all';

                let filteredCars = Object.entries(carData).filter(([id, car]) => {
                    if (car.type !== currentType) return false;

                    if (selectedBudget !== 'all') {
                        if (car.budgetRange !== selectedBudget) return false;
                    }

                    if (selectedBodyType !== 'all' && car.bodyType !== selectedBodyType) return false;

                    if (selectedFuelType !== 'all' && car.fuelType !== selectedFuelType) return false;

                    return true;
                }).map(([id, car]) => ({ id, ...car }));

                displayCars(filteredCars);
            }

            function displayCars(carsToDisplay) {
                filteredCarsContainer.innerHTML = '';

                if (carsToDisplay.length === 0) {
                    filteredCarsContainer.innerHTML = '<div class="no-results">No cars found matching your criteria. Please try different filters.</div>';
                    return;
                }

                carsToDisplay.forEach((car, index) => {
                    const carElement = document.createElement('div');
                    carElement.className = 'filtered-car';
                    carElement.style.animationDelay = `${index * 0.1}s`;
                    carElement.innerHTML = `
                        <a href="${car.link}" id="${car.id}">
                            <img src="${car.image}" alt="${car.name}">
                            <div class="filtered-car-details">
                                <h3 class="filtered-car-title">${car.name}</h3>
                                <p class="filtered-car-price">${car.price}</p>
                            </div>
                        </a>
                    `;
                    filteredCarsContainer.appendChild(carElement);
                });
            }
        }

                    const windowHeight = window.innerHeight;

                        section.style.transform = 'translateY(0)';
