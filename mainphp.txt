    // Function to compare two cars
    function compareCars(car1Id, car2Id) {
        const car1 = carDatabase[car1Id];
        const car2 = carDatabase[car2Id];
        
        // Update UI with car data
        document.getElementById('compareCar1Image').src = car1.image;
        document.getElementById('compareCar1Name').textContent = car1.name;

        document.getElementById('compareCar2Price').textContent = car2.price;
        
        // Generate comparison table
        const featuresContainer = document.getElementById('comparisonFeatures');
        featuresContainer.innerHTML = '';
        
        // Get all unique features from both cars
        const allFeatures = new Set([...Object.keys(car1.features), ...Object.keys(car2.features)]);
        
        allFeatures.forEach(feature => {
            const row = document.createElement('div');
            row.className = 'comparison-row';
            
            const featureName = document.createElement('div');
            featureName.className = 'comparison-feature-name';
            featureName.textContent = feature;
            row.appendChild(featureName);
            
            const car1Value = document.createElement('div');
            car1Value.className = 'comparison-feature-value';
            car1Value.textContent = car1.features[feature] || '-';
            row.appendChild(car1Value);
            
            const car2Value = document.createElement('div');
            car2Value.className = 'comparison-feature-value';
            car2Value.textContent = car2.features[feature] || '-';
            row.appendChild(car2Value);
            
            // Highlight better value (simple comparison for demo)
            if (car1.features[feature] && car2.features[feature]) {
                if (feature.toLowerCase().includes('power') || 
                    feature.toLowerCase().includes('torque') ||
                    feature.toLowerCase().includes('range') ||
                    feature.toLowerCase().includes('mileage') ||
                    feature.toLowerCase().includes('space')) {
                    const val1 = parseFloat(car1.features[feature]);
                    const val2 = parseFloat(car2.features[feature]);
                    
                    if (!isNaN(val1)) {
                        if (val1 > val2) {
                            car1Value.classList.add('comparison-highlight');
                        } else if (val2 > val1) {
                            car2Value.classList.add('comparison-highlight');
                        }
                    }
                }
            }
            
            featuresContainer.appendChild(row);
        });
        
        // Show results
        noComparison.style.display = 'none';
        comparisonData.style.display = 'block';
        comparisonResults.style.display = 'block';
    }
